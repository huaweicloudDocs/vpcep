    ```
    PUT https://{endpoint}/v1/{project_id}/vpc-endpoint-services/4189d3c2-8882-4871-a3c2-d380272eed88
    ```

    ```
    {
       "approval_enabled":true,
       "ports":[
                 {
                    "client_port":8081,
                    "server_port":22,
                    "protocol":"TCP"
                 },
                 {
                    "client_port":8082,
                    "server_port":23,
                    "protocol":"UDP"
                 }
               ]
    }
    ```
